// Ek dum obvious question hai ye.
/*
    Har stick to size-1 times break kar sakte hai.
    Har baar 1-1 size ka break karte rahenge.
    Last mein jab 1 size ka reh jaayega tab break possible nahi hoga.
*/
#include <bits/stdc++.h>
using namespace std;

int main() {
    // your code goes here
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector < int > nums(n);
        for (int i = 0; i < n; i++)
            cin >> nums[i];
        int ans = 0;
        for (int i = 0; i < n; i++)
            ans += nums[i] - 1;
        cout << ans << endl;
    }
}
