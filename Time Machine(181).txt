    Agar add karne par at least 2050 aa jaaye to apna kaam ban gaya.
*/

int main() {
    // your code goes here
    int t;
    cin >> t;   // Input lena.
    
    // Current year mein 25 add karne par at least 2050 aana chahiye.

}
